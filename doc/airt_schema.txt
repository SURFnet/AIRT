The following is the AIRT database schema. 

INCIDENTTYPES ( id, label )
    primary key: id

STATES ( id, label )
    primary key: id

STATUS ( id, label)
    primary key: id

CONSTITUENCIES ( id, label, name )
    primary key: id

USERS ( id, lastname, firstname, email, phone )
    primary key: id

ROLES ( id, label )
    primary key: id

INCIDENTS ( id, created, creator, updated, updatedby, state, status, type )
    primary key: id
    foreign key: creator references users(id)
                 updatedby references users(id)
                 state     references states(id)
                 status    references status(id)
                 type      references incidenttypes(id)


INCIDENT_ADDRESSES ( id, incident, address, added, addedby )
    primary key: id
    foreign key: incident references incidents(id)
                 address  referecens ipaddresses(id)
                 addedby  references users(id)

ROLE_ASSSIGNMENTS ( id, role, user )
    primary key: id
    foreign key: role references roles(id)
                 user references users(id)

IPADDRESSES ( id, ip, hostname, constituency, client )
    primary key: id
    foreign key: constituency references constituencies(id)
                 client references users(id)

CONSTITUENCY_CONTACTS ( id, constituency, user )
    primary key: id
    foreign key: constituency references constituencies(id)
                 user references users(id)

NETWORKS ( id, network, netmask, constituency )
    primary key: id
    foreign key: constituencies(id)

CREDENTIALS ( id, user, login, password, organizationalunit, certauth )
    primary key: id
    foreign key: user references users(id)

INCIDENT_COMMENTS ( id, incident, type, comment )
    primary key: id
    foreign key: incident references incidents(id)

IP_COMMENTS ( id, address, comment, added, addedby )
    primary key: id
    foreign key: address references ipaddresses(id)

INCIDENT_COMMENTS ( id, incident, comment, added, addedby )
    primary key: id
    foreign key: incidents references incidents(id)

USER_COMMENTS ( id, user, comment, added, addedby )
    primary key: id
    foreign key: user references users(id)
